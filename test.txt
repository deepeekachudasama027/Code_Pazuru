version: '3.7'

services:
    webapp-server:
      build:
        context: .
        dockerfile: Dockerfile
      image: topsy_turvy-server-img
      container_name: topsy_turvy-node-express
      volumes:
       - .:/usr/src/app
       - /usr/src/app/node_modules
      ports:
        - "5000:5000"
      depends_on:
        - mongo
      env_file: .env
      environment:
        - MONGO_URI=$MONGO_URI
    mongo:
      image: mongo
      container_name: topsy_turvy-mongodb
      ports:
        - "27017:27017"
